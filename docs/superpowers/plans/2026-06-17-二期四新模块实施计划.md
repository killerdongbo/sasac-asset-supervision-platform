```java
package com.sasac.platform.hr.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.sasac.platform.common.exception.BusinessException;
import com.sasac.platform.hr.dto.EmployeeCreateDTO;
import com.sasac.platform.hr.dto.EmployeeQueryDTO;
import com.sasac.platform.hr.entity.HrEmployee;
import com.sasac.platform.hr.mapper.HrEmployeeMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@RequiredArgsConstructor
public class HrEmployeeService {

    private final HrEmployeeMapper employeeMapper;

    @Transactional
    public HrEmployee create(EmployeeCreateDTO dto) {
        LambdaQueryWrapper<HrEmployee> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(HrEmployee::getTenantId, dto.getTenantId())
               .eq(HrEmployee::getEmployeeNo, dto.getEmployeeNo());
        if (employeeMapper.selectCount(wrapper) > 0) {
            throw new BusinessException("工号已存在：" + dto.getEmployeeNo());
        }

        HrEmployee employee = new HrEmployee();
        BeanUtils.copyProperties(dto, employee);
        if (employee.getStatus() == null) {
            employee.setStatus("ACTIVE");
        }
        employeeMapper.insert(employee);
        return employee;
    }

    public HrEmployee getById(Long id) {
        HrEmployee employee = employeeMapper.selectById(id);
        if (employee == null) {
            throw new BusinessException("人员不存在");
        }
        return employee;
    }

    public Page<HrEmployee> query(EmployeeQueryDTO query) {
        LambdaQueryWrapper<HrEmployee> wrapper = new LambdaQueryWrapper<>();

        if (query.getOrgId() != null) {
            wrapper.eq(HrEmployee::getOrgId, query.getOrgId());
        }
        if (query.getDeptId() != null) {
            wrapper.eq(HrEmployee::getDeptId, query.getDeptId());
        }
        if (query.getStatus() != null && !query.getStatus().isBlank()) {
            wrapper.eq(HrEmployee::getStatus, query.getStatus());
        }
        if (query.getEmploymentType() != null && !query.getEmploymentType().isBlank()) {
            wrapper.eq(HrEmployee::getEmploymentType, query.getEmploymentType());
        }
        if (query.getKeyword() != null && !query.getKeyword().isBlank()) {
            wrapper.and(w -> w
                    .like(HrEmployee::getName, query.getKeyword())
                    .or()
                    .like(HrEmployee::getEmployeeNo, query.getKeyword())
                    .or()
                    .like(HrEmployee::getPhone, query.getKeyword())
            );
        }
        wrapper.orderByDesc(HrEmployee::getId);

        return employeeMapper.selectPage(
                new Page<>(query.getPage(), query.getLimit()), wrapper);
    }

    @Transactional
    public HrEmployee update(Long id, HrEmployee update) {
        HrEmployee existing = getById(id);
        BeanUtils.copyProperties(update, existing, "id", "employeeNo", "tenantId",
                "createdAt", "updatedAt", "deleted");
        employeeMapper.updateById(existing);
        return employeeMapper.selectById(id);
    }

    @Transactional
    public void delete(Long id) {
        getById(id);
        employeeMapper.deleteById(id);
    }
}
```

```java
package com.sasac.platform.hr.controller;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.sasac.platform.common.response.ApiResponse;
import com.sasac.platform.hr.dto.EmployeeCreateDTO;
import com.sasac.platform.hr.dto.EmployeeQueryDTO;
import com.sasac.platform.hr.entity.HrEmployee;
import com.sasac.platform.hr.service.HrEmployeeService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/hr/employees")
@RequiredArgsConstructor
public class HrEmployeeController {

    private final HrEmployeeService employeeService;

    @PostMapping
    public ResponseEntity<ApiResponse<HrEmployee>> create(@Valid @RequestBody EmployeeCreateDTO dto) {
        HrEmployee created = employeeService.create(dto);
        return ResponseEntity.ok(ApiResponse.success(created));
    }

    @GetMapping("/{id}")
    public ResponseEntity<ApiResponse<HrEmployee>> getById(@PathVariable Long id) {
        return ResponseEntity.ok(ApiResponse.success(employeeService.getById(id)));
    }

    @GetMapping
    public ResponseEntity<ApiResponse<List<HrEmployee>>> query(@ModelAttribute EmployeeQueryDTO query) {
        Page<HrEmployee> page = employeeService.query(query);
        ApiResponse.PageMeta meta = ApiResponse.PageMeta.builder()
                .total(page.getTotal()).page((int) page.getCurrent()).limit((int) page.getSize()).build();
        return ResponseEntity.ok(ApiResponse.success(page.getRecords(), meta));
    }

    @PutMapping("/{id}")
    public ResponseEntity<ApiResponse<HrEmployee>> update(@PathVariable Long id, @RequestBody HrEmployee update) {
        return ResponseEntity.ok(ApiResponse.success(employeeService.update(id, update)));
    }

    @DeleteMapping("/{id}")
    public ResponseEntity<ApiResponse<Void>> delete(@PathVariable Long id) {
        employeeService.delete(id);
        return ResponseEntity.ok(ApiResponse.success(null));
    }
}
```

```java
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.util.List;

@Service
@RequiredArgsConstructor
public class HrContractService {

    private final HrContractMapper contractMapper;

    @Transactional
    public HrContract create(ContractCreateDTO dto) {
        HrContract contract = new HrContract();
        org.springframework.beans.BeanUtils.copyProperties(dto, contract);
        contract.setStatus("ACTIVE");
        contractMapper.insert(contract);
        return contract;
    }

    public HrContract getById(Long id) {
        HrContract contract = contractMapper.selectById(id);
        if (contract == null) throw new BusinessException("合同不存在");
        return contract;
    }

    public Page<HrContract> query(Long employeeId, Long tenantId, int page, int limit) {
        LambdaQueryWrapper<HrContract> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(HrContract::getTenantId, tenantId);
        if (employeeId != null) wrapper.eq(HrContract::getEmployeeId, employeeId);
        wrapper.orderByDesc(HrContract::getSignDate);
        return contractMapper.selectPage(new Page<>(page, limit), wrapper);
    }

    /** 查询30天内到期或已过期的合同 */
    public List<HrContract> findExpiring(Long tenantId, int days) {
        LambdaQueryWrapper<HrContract> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(HrContract::getTenantId, tenantId)
               .eq(HrContract::getStatus, "ACTIVE")
               .eq(HrContract::getIsUnlimited, false)
               .le(HrContract::getEndDate, LocalDate.now().plusDays(days));
        return contractMapper.selectList(wrapper);
    }

    @Transactional
    public HrContract update(Long id, HrContract update) {
        HrContract existing = getById(id);
        org.springframework.beans.BeanUtils.copyProperties(update, existing,
                "id", "tenantId", "createdAt", "updatedAt", "deleted");
        contractMapper.updateById(existing);
        return contractMapper.selectById(id);
    }

    @Transactional
    public void delete(Long id) {
        getById(id);
        contractMapper.deleteById(id);
    }
}
```

```java
package com.sasac.platform.decision.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.sasac.platform.common.exception.BusinessException;
import com.sasac.platform.decision.dto.*;
import com.sasac.platform.decision.entity.*;
import com.sasac.platform.decision.mapper.*;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.util.List;

@Service
@RequiredArgsConstructor
public class DecisionService {

    private final DecisionItemMapper itemMapper;
    private final DecisionMeetingMapper meetingMapper;
    private final DecisionResolutionMapper resolutionMapper;
    private final DecisionSupervisionMapper supervisionMapper;

    @Transactional
    public DecisionItem submitItem(DecisionItemCreateDTO dto) {
        DecisionItem item = new DecisionItem();
        org.springframework.beans.BeanUtils.copyProperties(dto, item);
        item.setStatus("DRAFT");
        item.setItemNo("DJ-" + System.currentTimeMillis());
        itemMapper.insert(item);
        return item;
    }

    public Page<DecisionItem> queryItems(Long tenantId, String itemType, String status, int page, int limit) {
        LambdaQueryWrapper<DecisionItem> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(DecisionItem::getTenantId, tenantId);
        if (itemType != null) wrapper.eq(DecisionItem::getItemType, itemType);
        if (status != null) wrapper.eq(DecisionItem::getStatus, status);
        wrapper.orderByDesc(DecisionItem::getId);
        return itemMapper.selectPage(new Page<>(page, limit), wrapper);
    }

    @Transactional
    public DecisionMeeting createMeeting(MeetingCreateDTO dto) {
        DecisionMeeting meeting = new DecisionMeeting();
        org.springframework.beans.BeanUtils.copyProperties(dto, meeting);
        meeting.setStatus("SCHEDULED");
        meeting.setMeetingNo("MT-" + System.currentTimeMillis());
        meetingMapper.insert(meeting);
        return meeting;
    }

    @Transactional
    public DecisionResolution createResolution(Long meetingId, ResolutionCreateDTO dto) {
        DecisionMeeting meeting = meetingMapper.selectById(meetingId);
        if (meeting == null) throw new BusinessException("会议不存在");

        DecisionResolution resolution = new DecisionResolution();
        org.springframework.beans.BeanUtils.copyProperties(dto, resolution);
        resolution.setMeetingId(meetingId);
        resolution.setTenantId(meeting.getTenantId());
        resolution.setResolutionNo("RES-" + System.currentTimeMillis());
        resolution.setStatus("APPROVED");
        resolutionMapper.insert(resolution);
        return resolution;
    }

    @Transactional
    public DecisionSupervision createSupervision(Long resolutionId, Long tenantId,
            String taskTitle, Long assigneeId, String assigneeName, LocalDate deadline) {
        DecisionSupervision supervision = new DecisionSupervision();
        supervision.setResolutionId(resolutionId);
        supervision.setTenantId(tenantId);
        supervision.setTaskTitle(taskTitle);
        supervision.setAssigneeId(assigneeId);
        supervision.setAssigneeName(assigneeName);
        supervision.setDeadline(deadline);
        supervision.setStatus("PENDING");
        supervisionMapper.insert(supervision);
        return supervision;
    }

    public List<DecisionSupervision> queryPendingSupervisions(Long tenantId) {
        LambdaQueryWrapper<DecisionSupervision> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(DecisionSupervision::getTenantId, tenantId)
               .eq(DecisionSupervision::getStatus, "PENDING");
        return supervisionMapper.selectList(wrapper);
    }
}
```

```java
// InvestService.java
public List<InvestEquityNode> buildEquityTree(Long projectId, Long tenantId) {
    List<InvestEquityNode> allNodes = equityNodeMapper.selectList(
        new LambdaQueryWrapper<InvestEquityNode>()
            .eq(InvestEquityNode::getTenantId, tenantId));
    // 构建树：递归查找根节点，按 parent_id 构建子节点列表
    return allNodes.stream()
        .filter(n -> n.getParentId() == null)
        .peek(n -> n.setChildren(buildChildren(n.getId(), allNodes)))
        .collect(Collectors.toList());
}
```

```java
// HrSalaryService.java
public HrSalary calculateMonthlySalary(Long employeeId, Long tenantId, int year, int month) {
    HrEmployee employee = employeeMapper.selectById(employeeId);
    if (employee == null) throw new BusinessException("人员不存在");

    // 获取考勤数据
    List<HrAttendance> attendances = attendanceMapper.selectList(
        new LambdaQueryWrapper<HrAttendance>()
            .eq(HrAttendance::getTenantId, tenantId)
            .eq(HrAttendance::getEmployeeId, employeeId)
            .between(HrAttendance::getAttDate,
                LocalDate.of(year, month, 1), LocalDate.of(year, month, 1).plusMonths(1).minusDays(1)));

    // 获取绩效数据（如存在）
    HrPerformance perf = performanceMapper.selectOne(
        new LambdaQueryWrapper<HrPerformance>()
            .eq(HrPerformance::getEmployeeId, employeeId)
            .eq(HrPerformance::getCycleYear, year)
            .eq(HrPerformance::getStatus, "CONFIRMED"));

    HrSalary salary = new HrSalary();
    salary.setTenantId(tenantId);
    salary.setEmployeeId(employeeId);
    salary.setSalaryYear(year);
    salary.setSalaryMonth(month);
    salary.setBaseSalary(new BigDecimal("8000"));  // 从薪酬方案读取
    salary.setPerformancePay(perf != null && perf.getFinalScore() != null
        ? BigDecimal.valueOf(perf.getFinalScore()).multiply(new BigDecimal("100")) : BigDecimal.ZERO);
    salary.setSocialInsurance(salary.getBaseSalary().multiply(new BigDecimal("0.105")));
    salary.setHousingFund(salary.getBaseSalary().multiply(new BigDecimal("0.12")));
    salary.setNetSalary(salary.getBaseSalary()
        .add(salary.getPerformancePay())
        .subtract(salary.getSocialInsurance())
        .subtract(salary.getHousingFund())
        .subtract(salary.getTax() != null ? salary.getTax() : BigDecimal.ZERO));
    salary.setStatus("CALCULATED");
    return salary;
}
```

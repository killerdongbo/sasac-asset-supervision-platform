package com.sasac.platform.asset.basic.controller;

import com.sasac.platform.asset.basic.entity.*;
import com.sasac.platform.asset.basic.mapper.*;
import com.sasac.platform.common.response.ApiResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequiredArgsConstructor
public class BasicDataController {

    private final LocationMapper locationMapper;
    private final SupplierMapper supplierMapper;
    private final MaintenanceProviderMapper providerMapper;
    private final AssetCategoryMapper categoryMapper;

    // ===== 存放位置 =====
    @GetMapping("/api/locations")
    public ApiResponse<List<Location>> listLocations() {
        return ApiResponse.success(locationMapper.selectList(null));
    }

    @PostMapping("/api/locations")
    public ApiResponse<Location> createLocation(@RequestBody Location location) {
        locationMapper.insert(location);
        return ApiResponse.success(location);
    }

    @PutMapping("/api/locations/{id}")
    public ApiResponse<Location> updateLocation(@PathVariable Long id, @RequestBody Location location) {
        location.setId(id);
        locationMapper.updateById(location);
        return ApiResponse.success(locationMapper.selectById(id));
    }

    @DeleteMapping("/api/locations/{id}")
    public ApiResponse<Void> deleteLocation(@PathVariable Long id) {
        locationMapper.deleteById(id);
        return ApiResponse.success(null);
    }

    // ===== 供应商 =====
    @GetMapping("/api/suppliers")
    public ApiResponse<List<Supplier>> listSuppliers() {
        return ApiResponse.success(supplierMapper.selectList(null));
    }

    @PostMapping("/api/suppliers")
    public ApiResponse<Supplier> createSupplier(@RequestBody Supplier supplier) {
        supplierMapper.insert(supplier);
        return ApiResponse.success(supplier);
    }

    @PutMapping("/api/suppliers/{id}")
    public ApiResponse<Supplier> updateSupplier(@PathVariable Long id, @RequestBody Supplier supplier) {
        supplier.setId(id);
        supplierMapper.updateById(supplier);
        return ApiResponse.success(supplierMapper.selectById(id));
    }

    @DeleteMapping("/api/suppliers/{id}")
    public ApiResponse<Void> deleteSupplier(@PathVariable Long id) {
        supplierMapper.deleteById(id);
        return ApiResponse.success(null);
    }

    // ===== 维保商 =====
    @GetMapping("/api/maintenance-providers")
    public ApiResponse<List<MaintenanceProvider>> listProviders() {
        return ApiResponse.success(providerMapper.selectList(null));
    }

    @PostMapping("/api/maintenance-providers")
    public ApiResponse<MaintenanceProvider> createProvider(@RequestBody MaintenanceProvider provider) {
        providerMapper.insert(provider);
        return ApiResponse.success(provider);
    }

    @PutMapping("/api/maintenance-providers/{id}")
    public ApiResponse<MaintenanceProvider> updateProvider(@PathVariable Long id, @RequestBody MaintenanceProvider provider) {
        provider.setId(id);
        providerMapper.updateById(provider);
        return ApiResponse.success(providerMapper.selectById(id));
    }

    @DeleteMapping("/api/maintenance-providers/{id}")
    public ApiResponse<Void> deleteProvider(@PathVariable Long id) {
        providerMapper.deleteById(id);
        return ApiResponse.success(null);
    }

    // ===== 资产分类 =====
    @GetMapping("/api/categories")
    public ApiResponse<List<AssetCategory>> listCategories() {
        return ApiResponse.success(categoryMapper.selectList(null));
    }

    @GetMapping("/api/categories/tree")
    public ApiResponse<List<AssetCategory>> treeCategories() {
        return ApiResponse.success(categoryMapper.selectList(
                new com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper<AssetCategory>()
                        .isNull(AssetCategory::getParentId)));
    }

    @PostMapping("/api/categories")
    public ApiResponse<AssetCategory> createCategory(@RequestBody AssetCategory category) {
        categoryMapper.insert(category);
        return ApiResponse.success(category);
    }
}

package com.sasac.platform.asset.ledger.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.sasac.platform.asset.ledger.dto.AssetCreateDTO;
import com.sasac.platform.asset.ledger.dto.AssetQueryDTO;
import com.sasac.platform.asset.ledger.entity.Asset;
import com.sasac.platform.asset.ledger.mapper.AssetMapper;
import com.sasac.platform.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.concurrent.atomic.AtomicInteger;

@Service
@RequiredArgsConstructor
public class AssetService {

    private final AssetMapper mapper;
    private static final AtomicInteger CODE_SEQ = new AtomicInteger(1);

    public Asset create(AssetCreateDTO dto) {
        if (dto.getAssetCode() != null && !dto.getAssetCode().isEmpty()) {
            Long count = mapper.selectCount(new LambdaQueryWrapper<Asset>()
                    .eq(Asset::getAssetCode, dto.getAssetCode()));
            if (count > 0) {
                throw new BusinessException("资产编码已存在: " + dto.getAssetCode());
            }
        }

        Asset asset = new Asset();
        BeanUtils.copyProperties(dto, asset);

        if (asset.getAssetCode() == null || asset.getAssetCode().isEmpty()) {
            String date = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd"));
            asset.setAssetCode("ZC-" + date + "-" + String.format("%04d", CODE_SEQ.incrementAndGet()));
        }
        if (asset.getCurrentValue() == null) {
            asset.setCurrentValue(asset.getOriginalValue());
        }
        if (asset.getAccumulatedDepreciation() == null) {
            asset.setAccumulatedDepreciation(BigDecimal.ZERO);
        }
        if (asset.getRegistrationDate() == null) {
            asset.setRegistrationDate(LocalDate.now());
        }
        if (asset.getQuantity() == null) {
            asset.setQuantity(1);
        }
        if (asset.getResidualRate() == null) {
            asset.setResidualRate(new BigDecimal("0.05"));
        }

        mapper.insert(asset);
        return asset;
    }

    public Asset getById(Long id) {
        Asset asset = mapper.selectById(id);
        if (asset == null) throw new BusinessException("资产不存在");
        return asset;
    }

    public List<Asset> query(AssetQueryDTO query) {
        Page<Asset> page = new Page<>(query.getPage(), query.getLimit());
        LambdaQueryWrapper<Asset> wrapper = new LambdaQueryWrapper<>();
        if (query.getOrgId() != null) wrapper.eq(Asset::getOrgId, query.getOrgId());
        if (query.getCategory() != null) wrapper.eq(Asset::getCategory, query.getCategory());
        if (query.getUseStatus() != null) wrapper.eq(Asset::getUseStatus, query.getUseStatus());
        if (query.getKeyword() != null) {
            wrapper.and(w -> w.like(Asset::getName, query.getKeyword())
                    .or().like(Asset::getAssetCode, query.getKeyword()));
        }
        wrapper.orderByDesc(Asset::getCreatedAt);
        return mapper.selectPage(page, wrapper).getRecords();
    }

    public Asset update(Long id, Asset update) {
        Asset existing = getById(id);
        BeanUtils.copyProperties(update, existing, "id", "assetCode", "tenantId",
                "createdAt", "updatedAt", "deleted");
        mapper.updateById(existing);
        return mapper.selectById(id);
    }

    public void delete(Long id) {
        getById(id);
        mapper.deleteById(id);
    }
}

package com.sasac.platform.supervision.approval.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.sasac.platform.common.exception.BusinessException;
import com.sasac.platform.supervision.approval.entity.*;
import com.sasac.platform.supervision.approval.mapper.*;
import lombok.RequiredArgsConstructor;
import lombok.SneakyThrows;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.*;

@Service
@RequiredArgsConstructor
public class ApprovalService {

    private final ApprovalDefMapper defMapper;
    private final ApprovalNodeMapper nodeMapper;
    private final ApprovalInstanceMapper instanceMapper;
    private final ObjectMapper objectMapper;

    public ApprovalDef createDef(ApprovalDef def) {
        defMapper.insert(def);
        return def;
    }

    public ApprovalNode addNode(ApprovalNode node) {
        nodeMapper.insert(node);
        return node;
    }

    @Transactional
    public ApprovalInstance startInstance(Long defId, Long bizId, String bizType, Long submitterId) {
        ApprovalDef def = defMapper.selectById(defId);
        if (def == null) throw new BusinessException("审批定义不存在");

        ApprovalInstance instance = new ApprovalInstance();
        instance.setDefId(defId);
        instance.setBizId(bizId);
        instance.setBizType(bizType);
        instance.setStatus("PENDING");
        instance.setCurrentNode(1);
        instance.setSubmitterId(submitterId);
        instance.setTenantId(def.getTenantId());
        instance.setNodeResults("[]");
        instanceMapper.insert(instance);
        return instance;
    }

    @Transactional
    @SneakyThrows
    public ApprovalInstance approve(Long instanceId, Long approverId,
                                     boolean approved, String remark) {
        ApprovalInstance instance = instanceMapper.selectById(instanceId);
        if (instance == null) throw new BusinessException("审批实例不存在");
        if (!"PENDING".equals(instance.getStatus())) {
            throw new BusinessException("当前状态不可审批: " + instance.getStatus());
        }

        List<Map<String, Object>> results = objectMapper.readValue(
                instance.getNodeResults(), new TypeReference<List<Map<String, Object>>>() {});

        Map<String, Object> nodeResult = new LinkedHashMap<>();
        nodeResult.put("node", instance.getCurrentNode());
        nodeResult.put("approverId", approverId);
        nodeResult.put("result", approved ? "APPROVED" : "REJECTED");
        nodeResult.put("remark", remark);
        nodeResult.put("time", LocalDateTime.now().toString());
        results.add(nodeResult);

        if (!approved) {
            instance.setStatus("REJECTED");
        } else {
            List<ApprovalNode> nodes = nodeMapper.selectList(
                    new LambdaQueryWrapper<ApprovalNode>()
                            .eq(ApprovalNode::getDefId, instance.getDefId())
                            .orderByAsc(ApprovalNode::getNodeOrder));
            int nextNode = instance.getCurrentNode() + 1;
            boolean hasNext = nodes.stream().anyMatch(n -> n.getNodeOrder() >= nextNode);
            if (hasNext) {
                instance.setCurrentNode(nextNode);
            } else {
                instance.setStatus("APPROVED");
            }
        }

        instance.setNodeResults(objectMapper.writeValueAsString(results));
        instanceMapper.updateById(instance);
        return instance;
    }

    public ApprovalInstance getInstance(Long id) {
        ApprovalInstance instance = instanceMapper.selectById(id);
        if (instance == null) throw new BusinessException("审批实例不存在");
        return instance;
    }

    public List<ApprovalInstance> getPendingApprovals(Long tenantId, String roleCode) {
        return instanceMapper.selectList(
                new LambdaQueryWrapper<ApprovalInstance>()
                        .eq(ApprovalInstance::getTenantId, tenantId)
                        .eq(ApprovalInstance::getStatus, "PENDING"));
    }

    public List<ApprovalInstance> getMyRequests(Long submitterId) {
        return instanceMapper.selectList(
                new LambdaQueryWrapper<ApprovalInstance>()
                        .eq(ApprovalInstance::getSubmitterId, submitterId)
                        .orderByDesc(ApprovalInstance::getCreatedAt));
    }
}

package com.sasac.platform.asset.inspection.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.sasac.platform.asset.inspection.entity.*;
import com.sasac.platform.asset.inspection.mapper.*;
import com.sasac.platform.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;

@Service
@RequiredArgsConstructor
public class InspectionService {

    private final InspectionTaskMapper taskMapper;
    private final InspectionRecordMapper recordMapper;
    private final InspectionAnomalyMapper anomalyMapper;

    public InspectionTask createTask(InspectionTask task) {
        task.setStatus("PENDING");
        task.setCompletedCount(0);
        taskMapper.insert(task);
        return task;
    }

    @Transactional
    public InspectionRecord recordInspection(InspectionRecord record) {
        InspectionTask task = taskMapper.selectById(record.getTaskId());
        if (task == null) throw new BusinessException("巡检任务不存在");

        record.setTenantId(task.getTenantId());
        recordMapper.insert(record);

        if (record.getIsNormal() != null && !record.getIsNormal()) {
            InspectionAnomaly anomaly = new InspectionAnomaly();
            anomaly.setTaskId(record.getTaskId());
            anomaly.setRecordId(record.getId());
            anomaly.setAssetId(record.getAssetId());
            anomaly.setAnomalyType(record.getAnomalyType());
            anomaly.setDescription(record.getDescription());
            anomaly.setStatus("OPEN");
            anomaly.setTenantId(task.getTenantId());
            anomalyMapper.insert(anomaly);
        }

        task.setCompletedCount(task.getCompletedCount() != null ? task.getCompletedCount() + 1 : 1);
        task.setStatus("IN_PROGRESS");
        taskMapper.updateById(task);
        return record;
    }

    public List<InspectionTask> getMyTasks(Long assigneeId) {
        return taskMapper.selectList(
                new LambdaQueryWrapper<InspectionTask>()
                        .eq(InspectionTask::getAssigneeId, assigneeId)
                        .orderByDesc(InspectionTask::getCreatedAt));
    }

    public List<InspectionRecord> getRecords(Long taskId) {
        return recordMapper.selectList(
                new LambdaQueryWrapper<InspectionRecord>()
                        .eq(InspectionRecord::getTaskId, taskId));
    }

    public List<InspectionAnomaly> getAnomalies(Long taskId) {
        return anomalyMapper.selectList(
                new LambdaQueryWrapper<InspectionAnomaly>()
                        .eq(InspectionAnomaly::getTaskId, taskId));
    }

    @Transactional
    public InspectionAnomaly resolveAnomaly(Long anomalyId, String resolution) {
        InspectionAnomaly anomaly = anomalyMapper.selectById(anomalyId);
        if (anomaly == null) throw new BusinessException("异常记录不存在");
        anomaly.setResolution(resolution);
        anomaly.setStatus("RESOLVED");
        anomalyMapper.updateById(anomaly);
        return anomaly;
    }

    @Transactional
    public void completeTask(Long taskId) {
        InspectionTask task = taskMapper.selectById(taskId);
        if (task == null) throw new BusinessException("任务不存在");
        task.setStatus("COMPLETED");
        taskMapper.updateById(task);
    }
}

package com.sasac.platform.asset.inventory.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.sasac.platform.asset.inventory.entity.*;
import com.sasac.platform.asset.inventory.mapper.*;
import com.sasac.platform.asset.ledger.entity.Asset;
import com.sasac.platform.asset.ledger.mapper.AssetMapper;
import com.sasac.platform.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;

@Service
@RequiredArgsConstructor
public class InventoryService {

    private final InventoryTaskMapper taskMapper;
    private final InventoryRecordMapper recordMapper;
    private final InventoryDiffMapper diffMapper;
    private final AssetMapper assetMapper;

    public InventoryTask createTask(InventoryTask task) {
        task.setStatus("PENDING");
        task.setCompletedCount(0);
        task.setDiffCount(0);
        taskMapper.insert(task);
        return task;
    }

    @Transactional
    public InventoryRecord recordInventory(InventoryRecord record) {
        InventoryTask task = taskMapper.selectById(record.getTaskId());
        if (task == null) throw new BusinessException("盘点任务不存在");

        Asset asset = assetMapper.selectById(record.getAssetId());
        if (asset != null) {
            record.setBookName(asset.getName());
            record.setBookLocation(asset.getLocationId() != null ? asset.getLocationId().toString() : "");
            record.setBookStatus(asset.getUseStatus());
        }
        record.setTenantId(task.getTenantId());
        recordMapper.insert(record);

        // 自动差异识别
        if (asset != null) {
            String diffType = null;
            if (record.getExists() != null && !record.getExists()) {
                diffType = "SHORTAGE";
            } else if (record.getActualLocation() != null &&
                    !record.getActualLocation().equals(record.getBookLocation())) {
                diffType = "WRONG_LOCATION";
            } else if (record.getActualStatus() != null &&
                    !record.getActualStatus().equals(record.getBookStatus())) {
                diffType = "STATUS_MISMATCH";
            }

            if (diffType != null) {
                InventoryDiff diff = new InventoryDiff();
                diff.setTaskId(record.getTaskId());
                diff.setRecordId(record.getId());
                diff.setAssetId(record.getAssetId());
                diff.setDiffType(diffType);
                diff.setBookValue("位置:" + record.getBookLocation() + " 状态:" + record.getBookStatus());
                diff.setActualValue("位置:" + record.getActualLocation() + " 状态:" + record.getActualStatus());

import java.time.LocalDateTime;

public class AlertScheduler {

    private final InspectionTaskMapper inspectionTaskMapper;
    private final MaintenanceRequestMapper maintenanceRequestMapper;
    private final AssetMapper assetMapper;
    private final AlertRecordMapper alertRecordMapper;

    @Scheduled(cron = "0 0 8 * * *")  // 每天8点执行
    public void scanInspectionOverdue() {
        List<InspectionTask> overdueTasks = inspectionTaskMapper.selectList(
                new LambdaQueryWrapper<InspectionTask>()
                        .lt(InspectionTask::getEndDate, LocalDate.now())
                        .ne(InspectionTask::getStatus, "COMPLETED"));
        for (InspectionTask task : overdueTasks) {
            AlertRecord alert = new AlertRecord();
            alert.setAlertType("INSPECTION_OVERDUE");
            alert.setTitle("巡检任务超期: " + task.getTaskName());
            alert.setContent("巡检任务截止日期 " + task.getEndDate() + " 已超期，当前状态: " + task.getStatus());
            alert.setStatus("ACTIVE");
            alert.setTenantId(task.getTenantId());
            alert.setRefId(task.getId());
            alertRecordMapper.insert(alert);
        }
        log.info("巡检超期扫描完成, 发现{}条超期任务", overdueTasks.size());
    }

    @Scheduled(cron = "0 0 8 * * *")
    public void scanIdleAssets() {
        LocalDate threshold = LocalDate.now().minusDays(90);
        List<Asset> idleAssets = assetMapper.selectList(
                new LambdaQueryWrapper<Asset>()
                        .eq(Asset::getUseStatus, "IDLE")
                        .lt(Asset::getUpdatedAt, threshold));
        for (Asset asset : idleAssets) {
            AlertRecord alert = new AlertRecord();
            alert.setAlertType("IDLE_ASSET");
            alert.setTitle("闲置资产预警: " + asset.getName());
            alert.setContent("资产 " + asset.getAssetCode() + " 已闲置超过90天，建议评估调拨或处置");
            alert.setStatus("ACTIVE");
            alert.setTenantId(asset.getTenantId());
            alert.setRefId(asset.getId());
            alertRecordMapper.insert(alert);
        }
        log.info("闲置资产扫描完成, 发现{}条闲置资产", idleAssets.size());
    }
}

import org.aspectj.lang.annotation.*;
import org.springframework.stereotype.Component;
import java.time.LocalDateTime;

@Slf4j
@Aspect
@Component
@RequiredArgsConstructor
public class OperationLogAspect {

    private final OperationLogMapper logMapper;
    private final ObjectMapper objectMapper;

    @Around("@annotation(com.sasac.platform.supervision.audit.aspect.Auditable)")
    @SneakyThrows
    public Object logOperation(ProceedingJoinPoint joinPoint) {
        Auditable annotation = joinPoint.getTarget().getClass()
                .getMethod(joinPoint.getSignature().getName(),
                        ((org.aspectj.lang.reflect.MethodSignature)
                                joinPoint.getSignature()).getParameterTypes())
                .getAnnotation(Auditable.class);
        if (annotation == null) return joinPoint.proceed();

        Object[] args = joinPoint.getArgs();
        String beforeData = args.length > 0 ? objectMapper.writeValueAsString(args[0]) : "{}";

        Object result = joinPoint.proceed();

        OperationLog log = new OperationLog();
        log.setAction(annotation.action());
        log.setTargetType(annotation.targetType());
        log.setBeforeData(beforeData);
        log.setAfterData(result != null ? objectMapper.writeValueAsString(result) : "{}");
        log.setTenantId(TenantContext.getCurrentTenant());
        log.setCreatedAt(LocalDateTime.now());
        logMapper.insert(log);

        return result;
    }

package com.sasac.platform.supervision.audit.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.sasac.platform.asset.ledger.entity.AssetChangeLog;
import com.sasac.platform.asset.ledger.mapper.AssetChangeLogMapper;
import com.sasac.platform.supervision.audit.entity.OperationLog;
import com.sasac.platform.supervision.audit.mapper.OperationLogMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import java.util.*;
import java.util.stream.Collectors;
import java.util.stream.Stream;

@Service
@RequiredArgsConstructor
public class AuditService {

    private final OperationLogMapper logMapper;
    private final AssetChangeLogMapper changeLogMapper;

    public List<OperationLog> getLogs(String targetType, Long targetId) {
        return logMapper.selectList(
                new LambdaQueryWrapper<OperationLog>()
                        .eq(OperationLog::getTargetType, targetType)
                        .eq(OperationLog::getTargetId, targetId)
                        .orderByDesc(OperationLog::getCreatedAt));
    }

    public List<Map<String, Object>> getAssetLifecycle(Long assetId) {
        List<AssetChangeLog> changes = changeLogMapper.selectList(
                new LambdaQueryWrapper<AssetChangeLog>()
                        .eq(AssetChangeLog::getAssetId, assetId)
                        .orderByAsc(AssetChangeLog::getCreatedAt));

        List<OperationLog> ops = logMapper.selectList(
                new LambdaQueryWrapper<OperationLog>()
                        .eq(OperationLog::getTargetType, "Asset")
                        .eq(OperationLog::getTargetId, assetId)
                        .orderByAsc(OperationLog::getCreatedAt));

        return Stream.concat(
                changes.stream().map(c -> {
                    Map<String, Object> m = new LinkedHashMap<>();
                    m.put("type", "CHANGE");
                    m.put("time", c.getCreatedAt());
                    m.put("field", c.getChangeField());
                    m.put("details", c.getBeforeValue() + " → " + c.getAfterValue());
                    return m;
                }),
                ops.stream().map(o -> {
                    Map<String, Object> m = new LinkedHashMap<>();
                    m.put("type", "OPERATION");
                    m.put("time", o.getCreatedAt());
                    m.put("field", o.getAction());
                    m.put("details", o.getBeforeData() + " → " + o.getAfterData());
                    return m;
                })
        ).sorted(Comparator.comparing(m -> ((java.time.LocalDateTime) m.get("time"))))
                .collect(Collectors.toList());
    }
}
